Technical Depth

Protocol-first. Mathematically grounded.

Every intent flows through a versioned proto contract. Every locator resolves from a declarative JSON artifact at runtime.

ptom.protoExecuteIntent RPC
syntax = "proto3";
package ptom;

message IntentRequest {
  string actionId       = 1; // NAVIGATE · CLICK · TYPE
  string targetSelector = 2; // logical key or raw
  string platform       = 3; // playwright:0 · appium:1
}

message IntentResponse {
  string status       = 1; // PASS | FAIL
  string payload      = 2;
  string errorMessage = 3;
}

service ActionService {
  rpc ExecuteIntent (IntentRequest)
    returns (IntentResponse);
}
